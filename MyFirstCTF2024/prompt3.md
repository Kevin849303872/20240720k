# ChatGPT
```c
#include <stdio.h>

int main() {
    int i, j;

    // 打印標題
    printf("  乘法表\n");
    printf("--------\n");

    // 打印乘法表格
    printf("  ");
    for (i = 1; i <= 9; i++) {
        printf("%3d", i); // 標題列（1 到 9）
    }
    printf("\n");

    for (i = 1; i <= 9; i++) {
        printf("%2d", i); // 行首（1 到 9）

        for (j = 1; j <= 9; j++) {
            printf("%3d", i * j); // 計算並輸出乘法結果
        }

        printf("\n");
    }

    return 0;
}
```
